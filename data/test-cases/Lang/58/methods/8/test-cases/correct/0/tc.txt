@Test public void testToDouble() { assertEquals(0.0d, NumberUtils.toDouble("0.0"), 0.0d); assertEquals(1.0d, NumberUtils.toDouble("1.0"), 0.0d); assertEquals(-1.0d, NumberUtils.toDouble("-1.0"), 0.0d); assertEquals(Double.POSITIVE_INFINITY, NumberUtils.toDouble("+Infinity"), 0.0d); assertEquals(Double.NEGATIVE_INFINITY, NumberUtils.toDouble("-Infinity"), 0.0d); assertEquals(Double.NaN, NumberUtils.toDouble("NaN"), 0.0d); }
