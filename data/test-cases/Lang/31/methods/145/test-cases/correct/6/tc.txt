@Test public void testStartsWithAny() { assertEquals(false, StringUtils.startsWithAny(null, new String[]{})); assertEquals(false, StringUtils.startsWithAny("", new String[]{})); assertEquals(false, StringUtils.startsWithAny("", new String[]{""})); assertEquals(true, StringUtils.startsWithAny("abc", new String[]{"abc"})); assertEquals(true, StringUtils.startsWithAny("abc", new String[]{"a", "b"})); assertEquals(true, StringUtils.startsWithAny("abc", new String[]{"a", "b"})); assertEquals(true, StringUtils.startsWithAny("abc", new String[]{"a", "b", "c"})); }
