@Test public void testIsSet() { String[] values = new String[]{"one", "two", "three", "four"}; Map<String, Integer> mapping = new HashMap<String, Integer>(); mapping.put("one", 1); mapping.put("two", 2); mapping.put("three", 3); mapping.put("four", 4); CSVRecord record = new CSVRecord(values, mapping, "comment", 1); assertTrue(record.isSet("one")); assertTrue(record.isSet("two")); assertTrue(record.isSet("three")); assertFalse(record.isSet("four")); }
