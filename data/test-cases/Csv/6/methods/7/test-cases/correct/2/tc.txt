@Test public void testIsSet() { final String[] values = new String[]{"one", "two", "three", "four"}; final Map<String, Integer> mappings = new HashMap<String, Integer>(); mappings.put("one", 1); mappings.put("two", 2); mappings.put("three", 3); mappings.put("four", 4); final CSVRecord record = new CSVRecord(values, mappings, "comment", 1); assertTrue(record.isSet("one")); assertTrue(record.isSet("two")); assertTrue(record.isSet("three")); assertFalse(record.isSet("four")); }
