@Test public void testIsSet() { final String[] values = new String[]{"one", "two", "three"}; final Map<String, Integer> mapping = new HashMap<String, Integer>(); mapping.put("one", 1); mapping.put("two", 2); mapping.put("three", 3); final CSVRecord record = new CSVRecord(values, mapping, "comment", 1); assertTrue(record.isSet("one")); assertTrue(record.isSet("two")); assertFalse(record.isSet("three")); }
