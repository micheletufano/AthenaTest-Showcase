@Test public void testIterator() { final String[] values = new String[]{"one", "two", "three", "four", "five"}; final Map<String, Integer> mapping = new HashMap<String, Integer>(); mapping.put("one", 1); mapping.put("two", 2); mapping.put("three", 3); mapping.put("four", 4); mapping.put("five", 5); final CSVRecord record = new CSVRecord(values, mapping, "comment", 1); final Iterator<String> iterator = record.iterator(); assertTrue(iterator.hasNext()); assertEquals("one", iterator.next()); assertEquals("two", iterator.next()); assertEquals("three", iterator.next()); assertEquals("four", iterator.next()); assertEquals("five", iterator.next()); assertFalse(iterator.hasNext()); }
