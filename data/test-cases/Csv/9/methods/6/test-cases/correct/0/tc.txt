@Test public void testIsMapped() { final String[] values = new String[]{"one", "two", "three", "four"}; final Map<String, Integer> mapping = new HashMap<String, Integer>(); mapping.put("one", 1); mapping.put("two", 2); mapping.put("three", 3); mapping.put("four", 4); final CSVRecord record = new CSVRecord(values, mapping, null, -1); assertTrue(record.isMapped("one")); assertTrue(record.isMapped("two")); assertTrue(record.isMapped("three")); assertTrue(record.isMapped("four")); }
