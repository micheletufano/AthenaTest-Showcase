@Test public void testGetDomainAxisIndex() { CategoryAxis xAxis = new CategoryAxis("X"); CategoryAxis yAxis = new CategoryAxis("Y"); CategoryPlot plot = new CategoryPlot(); assertEquals(-1, plot.getDomainAxisIndex(xAxis)); assertEquals(-1, plot.getDomainAxisIndex(yAxis)); plot.setDomainAxis(1, xAxis); assertEquals(1, plot.getDomainAxisIndex(xAxis)); assertEquals(-1, plot.getDomainAxisIndex(yAxis)); }
